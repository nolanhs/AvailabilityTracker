create database dbesp32;

use dbesp32;

create table tblUsers (
	userID INT auto_increment,
    Email varchar(255) NOT NULL unique,
    passwordHash varchar(255) NOT NULL,
    Role varchar(20) NOT NULL DEFAULT 'viewer',
    Major varchar(50) NOT NULL,
    primary key (userID)
);

create table tblBuildings (
	buildingID INT auto_increment,
    buildingName varchar(255) NOT NULL,
    buildingCode varchar(32) NOT NULL,
    primary key (buildingID)
);

create table tblstudyRooms (
	roomID INT auto_increment,
    buildingID INT NOT NULL,
    roomName varchar(255) NOT NULL,
    floorNumber INT,
    scannerID varchar (64),
    createdAt datetime NOT NULL default current_timestamp,
    primary key (roomID)
);

create table tblroomStatus (
	roomID INT NOT NULL,
    isOccupied BOOLEAN NOT NULL,
    enteredAt datetime,
    lastSeenat datetime,
    updatedAt datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
    primary key(roomID)
);

create table tblRoomhistory (
	historyID INT auto_increment,
    roomID INT NOT NULL,
    isOccupied BOOLEAN NOT NULL,
    startedAt DATETIME NOT NULL DEFAULT current_timestamp,
    endedAt datetime,
    Cause varchar(255),
    primary key(historyID)
);

-- last 7 days history
SELECT *
FROM tblRoomhistory
WHERE roomID = 1
  AND (
    startedAt >= (NOW() - INTERVAL 7 DAY)
    OR endedAt IS NULL
    OR endedAt >= (NOW() - INTERVAL 7 DAY)
  )
ORDER BY startedAt;

-- how long room was occupied for 7 days
SELECT 
    SUM(
        TIMESTAMPDIFF(
            MINUTE,
            startedAt,
            COALESCE(endedAt, NOW())
        )
    ) AS OccupiedMinutes
FROM tblRoomhistory
WHERE roomID = 1
  AND isOccupied = TRUE
  AND startedAt >= (NOW() - INTERVAL 7 DAY);

-- delete last 7 days
DELETE FROM tblRoomhistory
WHERE endedAt IS NOT NULL
  AND endedAt < (NOW() - INTERVAL 7 DAY);
